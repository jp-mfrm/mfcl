## Controlled Dropdown

<Playground>
  <Component initialState={{ value: { name: '1', value: '1' }, isOpen: false }}>
    {({ setState, state }) => (
      <Row>
        <Dropdown
          open={state.isOpen}
          onChange={(option) => setState({ value: option })}
          options={[
            { name: '1', value: '1' },
            { name: '2', value: '2' },
            { name: '3', value: '3' },
            { name: '4', value: '4' },
            { name: '5', value: '5', disabled: true },
            { name: '6', value: '6' }
          ]}
          value={state.value}
        />
        <Button btnType="primary" onClick={() => setState({ isOpen: !state.isOpen })} size="lg">
          {state.isOpen ? 'Close' : 'Open'} me
        </Button>
      </Row>
    )}
  </Component>
</Playground>
